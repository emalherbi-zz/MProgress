Basic usage

Simply call `start()` and `done()` to control the progress bar.

MProgress.start();
MProgress.done();

Title 

MProgress.configure({
  title: 'Example Title...'
});

MProgress.start();
window.setTimeout(function() {
  MProgress.done();
}, 5000);

Inc 

MProgress.configure({
  title: 'Example Inc...',
  progressInc : 10
});

MProgress.start();
window.setTimeout(function() {
  MProgress.done();
}, 5000);

Update Progress (3 seconds) 

MProgress.configure({
  title: 'Example Progress Update...',
  progressUpdate : 3 // 3 seconds
});

MProgress.start();
window.setTimeout(function() {
  MProgress.done();
}, 10000);

Timeout Progress (30 seconds) 

MProgress.configure({
  title: 'Example Timeout...',
  progressTimeout : 30
});

MProgress.start();

Add Event Done 

MProgress.configure({
  title: 'Example Add Event Done...'
});

MProgress.start();

MProgress.eventDone(function() {
  alert( 'Event Done !!!' );
});

window.setTimeout(function() {
  MProgress.done();
}, 5000);

Faster 

MProgress.configure({
  title: 'Update Progress + Inc (Faster)...',
  progressInc : 1,
  progressUpdate : 0.1,
  progressTimeout : 30
});

MProgress.start();
window.setTimeout(function() {
  MProgress.done();
}, 10000);
